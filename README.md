# 1. Clone the repository (only needed once)
git clone https://github.com/USERNAME/REPOSITORY.git
cd REPOSITORY

# 2. Create a new branch (optional but recommended)
git checkout -b my-edit-branch

# 3. Make your edits
# (edit files using your editor, e.g. VS Code, nano, vim)

# 4. Stage your changes
git add .

# 5. Commit the changes
git commit -m "Describe the update you made"

# 6. Push the changes to GitHub
git push -u origin my-edit-branch

# 7. If you want to merge into main (from GitHub UI)
# Go to GitHub → Pull Requests → Create Pull Request → Merge
